a = int(input("Enter a number: "))

result = []
odd_count = a if a % 2 != 0 else a - 1      # this line reduces one number if input is even
                                            # if odd it will keep same 
for i in range(1, odd_count + 1):
    result.append(2 * i - 1)
    
print("Output: ", ", ".join(map(str, result)))
