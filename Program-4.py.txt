def count_multiples(numbers):
    result = {}                      #dictionary to store results
    
    for i in range(1, 10):
        result[i] = sum(1 for n in numbers if n % i == 0)   #sums values if it is divisible by a number within given list
            
    return result    # returns result
    
numbers =  [1,2,8,9,12,46,76,82,15,20,30]
print("Output: ", count_multiples(numbers))
